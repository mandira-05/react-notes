import React from 'react'
import { BrowserRouter as Router, Link, Route } from 'react-router-dom'

  return (
    <div className="App">

        <Link to="/home" >Home</Link>

        <Link to="/about" >About</Link>

  );
}

    <div>

    <div>
      <h1>About Page</h1>

    </div>
  )
}
